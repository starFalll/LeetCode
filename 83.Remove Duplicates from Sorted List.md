### 83.Remove Duplicates from Sorted List

Given a sorted linked list, delete all duplicates such that each element appear only *once*.

**Example 1:**

```
Input: 1->1->2
Output: 1->2


```

**Example 2:**

```
Input: 1->1->2->3->3
Output: 1->2->3
```

```cpp
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode* deleteDuplicates(ListNode* head) {
        if(!head||!(head->next))
            return head;
        ListNode* front=head;
        ListNode* p=head->next;
        while(p){
            if(p->val==front->val){
                front->next=p->next;
                p=front->next;
            }
            else{
                front=front->next;
                p=p->next;
            }
        }
        return head;
    }
};
```
